module Quot (A : Set)(_∼_ : A → A → Set) where
  ⟪_⟫ : A → Subset A -- A → A → Set
  ⟪ a ⟫ = λ b → b ∼ a

  A/∼ : Set₁
  A/∼ = undefined


_/_ : (A : Set) → (A → A → Set) → Set₁
A / _∼_ = Quot.A/∼ A _∼_
